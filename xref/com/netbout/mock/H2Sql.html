27  package com.netbout.mock;
28  
29  import com.jcabi.aspects.Immutable;
30  import com.jcabi.aspects.Loggable;
31  import com.jcabi.jdbc.JdbcSession;
32  import java.io.File;
33  import java.io.IOException;
34  import java.sql.SQLException;
35  import javax.sql.DataSource;
36  import lombok.EqualsAndHashCode;
37  import lombok.ToString;
38  import org.h2.jdbcx.JdbcDataSource;
39  
40  /**
41   * Mock base.
42   *
43   * @author Yegor Bugayenko (yegor@teamed.io)
44   * @version $Id: 2ed9c36553be450d206e41bbeed348afe05b61d3 $
45   * @since 2.0
46   */
47  @Immutable
48  @ToString
49  @Loggable(Loggable.DEBUG)
50  @EqualsAndHashCode(of = "file")
51  final class H2Sql implements Sql {
52  
53      /**
54       * File with H2 database.
55       */
56      private final transient String file;
57  
58      /**
59       * Public ctor.
60       * @throws IOException If fails
61       */
62      H2Sql() throws IOException {
63          this.file = filename();
64      }
65  
66      @Override
67      public DataSource source() {
68          return H2Sql.source(this.file);
69      }
70  
71      /**
72       * Return the JdbcDataSource working file name.
73       * @return The JdbcDataSource file name
74       * @throws IOException If fails
75       */
76      private static String filename() throws IOException {
77          final File tmp = File.createTempFile("netbout-", ".h2");
78          tmp.deleteOnExit();
79          final String result = tmp.getAbsolutePath();
80          new File(String.format("%s.mv.db", result)).deleteOnExit();
81          final String[] stmts = {
82              // @checkstyle LineLength (5 lines)
83              "CREATE TABLE alias (name VARCHAR, urn VARCHAR, photo VARCHAR, locale VARCHAR, email VARCHAR)",
84              "CREATE TABLE bout (number BIGINT AUTO_INCREMENT, title VARCHAR, date TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP)",
85              "CREATE TABLE message (number BIGINT AUTO_INCREMENT, bout BIGINT, text VARCHAR, author VARCHAR, date TIMESTAMP DEFAULT CURRENT_TIMESTAMP)",
86              "CREATE TABLE attachment (name VARCHAR, bout BIGINT, data VARCHAR, author VARCHAR, ctype VARCHAR, etag VARCHAR, date TIMESTAMP DEFAULT CURRENT_TIMESTAMP)",
87              "CREATE TABLE friend (alias VARCHAR, bout BIGINT, subscription INTEGER )",
88          };
89          final JdbcSession session = new JdbcSession(H2Sql.source(result));
90          for (final String stmt : stmts) {
91              try {
92                  session.sql(stmt).execute();
93              } catch (final SQLException ex) {
94                  throw new IOException(ex);
95              }
96          }
97          return result;
98      }
99  
100     /**
101      * Returns the JdbcDataSource created on the given file.
102      * @param file The name of the file for the JdbcDataSource
103      * @return The JdbcDataSource created
104      */
105     private static DataSource source(final String file) {
106         final JdbcDataSource src = new JdbcDataSource();
107         src.setURL(String.format("jdbc:h2:%s", file));
108         src.setUser("");
109         src.setPassword("");
110         return src;
111     }
112 }
