69  /**
70   * Dynamo messages.
71   * @todo #603:30min Refactor DyMessages class to avoid suppressing of
72   *  PMD.TooManyMethods and PMD.ExcessiveImports warnings. for example

153      * Ctor.
154      * @param reg Region
155      * @param num Bout number
156      * @param slf Self alias

158      * @param strt Start message number
159      * @since 2.7.1
160      * @checkstyle ParameterNumberCheck (5 lines)
161      */
162     DyMessages(final Region reg, final long num, final String slf,
163         final Counter ctr, final long strt) {
164         this.region = reg;
165         this.bout = num;
166         this.self = slf;
167         this.counter = ctr;
168         this.start = strt;
169     }

171     @Override
172     public void post(final String text) throws IOException {
173         final String clean = StringUtils.stripEnd(text, null);
174         if (clean.isEmpty()) {
175             throw new Messages.BrokenPostException(
176                 "empty message content is not allowed"
177             );
178         }
179         if (clean.length() > Tv.TEN * Tv.THOUSAND) {
180             throw new Messages.BrokenPostException(
181                 "message is too big, break it into parts or upload attachment"
182             );
183         }
184         final long number = this.counter.incrementAndGet(1L);
185         this.region.table(DyMessages.TBL).put(
186             new Attributes()
187                 .with(DyMessages.HASH, this.bout)
188                 .with(DyMessages.RANGE, number)
189                 .with(DyMessages.ATTR_TEXT, clean)
190                 .with(DyMessages.ATTR_ALIAS, this.self)
191                 .with(DyMessages.ATTR_DATE, System.currentTimeMillis())
192         );
193         this.updated();
194         Logger.info(this, "posted to #%d by @%s", this.bout, this.self);
195     }

197     // @todo #1094:30min HttpException seems not really appropriate here
198     //  in database layer. I think we have to throw something like
199     //  BoutNotFoundException and process its somewhere in web layer.
200     //  Pay attention that we should also change a test
201     //  DyMessagesITCase.exceptionIfBoutNotFound() which tests a presence
202     //  of HttpException.
203     @Override
204     public long unread() throws IOException {
205         final Iterator<Item> iterator = this.region.table(DyFriends.TBL)
206             .frame()
207             .through(
208                 new QueryValve()
209                     .withLimit(1)
210                     .withAttributesToGet(DyFriends.ATTR_UNREAD)
211             )
212             .where(DyFriends.HASH, Conditions.equalTo(this.bout))
213             .where(DyFriends.RANGE, Conditions.equalTo(this.self))
214             .iterator();
215         if (!iterator.hasNext()) {
216             throw new HttpException(
217                 HttpURLConnection.HTTP_NOT_FOUND,
218                 new Inbox.BoutNotFoundException(this.bout)
219             );
220         }
221         final Item item = iterator.next();
222         final long unread;
223         if (item.has(DyFriends.ATTR_UNREAD)) {
224             unread = Long.parseLong(item.get(DyFriends.ATTR_UNREAD).getN());
225         } else {
226             unread = 0L;
227         }
228         return unread;
229     }

231     @Override
232     public Pageable<Message> jump(final long number) {
233         return new DyMessages(
234             this.region, this.bout, this.self, this.counter, number
235         );
236     }

238     @Override
239     public Iterable<Message> iterate() {
240         if (this.start != Long.MAX_VALUE) {
241             this.seen();
242         }
243         return Iterables.transform(
244             this.region.table(DyMessages.TBL)
245                 .frame()
246                 .through(
247                     new QueryValve()
248                         .withScanIndexForward(false)
249                         .withLimit(Messages.PAGE)
250                         .withAttributesToGet(
251                             DyMessages.ATTR_TEXT,
252                             DyMessages.ATTR_ALIAS,
253                             DyMessages.ATTR_DATE
254                         )
255                 )
256                 .where(DyMessages.HASH, Conditions.equalTo(this.bout))
257                 .where(
258                     DyMessages.RANGE,
259                     new Condition()
260                         .withComparisonOperator(ComparisonOperator.LT)
261                         .withAttributeValueList(
262                             new AttributeValue().withN(
263                                 Long.toString(this.start)
264                             )
265                         )
266                 ),
267             new Function<Item, Message>() {
268                 @Override
269                 public Message apply(final Item item) {
270                     return new DyMessage(item);
271                 }
272             }
273         );
274     }
275 
276     @Override
277     public Iterable<Message> search(final String term) throws IOException {
278         final List<Message> result = new ArrayList<>(16);
279         for (final Message message : this.iterate()) {
280             if (message.text().contains(term)) {
281                 result.add(message);
282             }
283         }
284         return result;
285     }

287     /**
288      * It was updated just now.
289      */
290     @Async
291     private void updated() {
292         final String alias = this.self;
293         Iterables.all(
294             this.region.table(DyFriends.TBL).frame()
295                 .through(new QueryValve())
296                 .where(DyFriends.HASH, Conditions.equalTo(this.bout)),
297             // @checkstyle AnonInnerLengthCheck (50 lines)
298             new Predicate<Item>() {
299                 @Override
300                 public boolean apply(final Item input) {
301                     AttributeUpdates updates = new AttributeUpdates().with(
302                         DyFriends.ATTR_UPDATED,
303                         System.currentTimeMillis()
304                     );
305                     try {
306                         if (!input.get(DyFriends.RANGE).getS().equals(alias)) {
307                             updates = updates.with(
308                                 DyFriends.ATTR_UNREAD,
309                                 new AttributeValueUpdate()
310                                     .withAction(AttributeAction.ADD)
311                                     .withValue(new AttributeValue().withN("1"))
312                             );
313                         }
314                         input.put(updates);
315                     } catch (final IOException ex) {
316                         throw new IllegalStateException(ex);
317                     }
318                     return true;
319                 }
320             }
321         );
322     }
323 
324     /**
325      * It was seen just now.
326      */
327     @Async
328     private void seen() {
329         Iterables.all(
330             this.region.table(DyFriends.TBL).frame()
331                 .through(new QueryValve())
332                 .where(DyFriends.HASH, Conditions.equalTo(this.bout))
333                 .where(DyFriends.RANGE, this.self),
334             new Predicate<Item>() {
335                 @Override
336                 public boolean apply(final Item input) {
337                     try {
338                         input.put(
339                             new AttributeUpdates().with(
340                                 DyFriends.ATTR_UNREAD, 0L
341                             )
342                         );
343                     } catch (final IOException ex) {
344                         throw new IllegalStateException(ex);
345                     }
346                     return true;
347                 }
348             }
349         );
350     }

352     /**
353      * Sttc counter.
354      * @return Counter
355      */
356     @Cacheable(forever = true)
357     private static Counter sttc() {
358         try {
359             return new CdSttc(
360                 new ReSttc(
361                     RtSttc.make(
362                         URN.create(Manifests.read("Netbout-SttcUrn")),
363                         Manifests.read("Netbout-SttcToken")
364                     )
365                 )
366             ).counters().get("nb-message");
367         } catch (final IOException ex) {
368             throw new IllegalStateException(ex);
369         }
370     }
