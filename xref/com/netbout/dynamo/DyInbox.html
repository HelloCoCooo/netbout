60  /**
61   * Dynamo inbox.
62   * @todo #603:30min Refactor DyInbox class to avoid suppressing of
63   *  PMD.TooManyMethods and PMD.ExcessiveImports warnings. for example
64   *  there are some private static methods there, those could be easily
65   *  extracted.
66   * @author Yegor Bugayenko (yegor@teamed.io)
67   * @version $Id: 5b48ea535e98a21c8d5a356237d0469d7b45f9e1 $
68   * @since 2.0
69   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
70   */
71  @Immutable
72  @Loggable(Loggable.DEBUG)
73  @ToString(of = "self")
74  @EqualsAndHashCode(of = { "counter", "region", "self", "since" })
75  @SuppressWarnings({ "PMD.TooManyMethods", "PMD.ExcessiveImports" })
76  final class DyInbox implements Inbox {
77  
78      /**
79       * Counter with bout number.
80       */
81      private final transient Counter counter;
82  
83      /**
84       * Region we're in.
85       */
86      private final transient Region region;
87  
88      /**
89       * Alias of myself.
90       */
91      private final transient String self;
92  
93      /**
94       * Start moment.
95       */
96      private final transient long since;
97  
98      /**
99       * Ctor.
100      * @param reg Region we're in
101      * @param slf My alias
102      */
103     DyInbox(final Region reg, final String slf) {
104         this(reg, slf, DyInbox.sttc(), Inbox.NEVER);
105     }
106 
107     /**
108      * Ctor.
109      * @param reg Region we're in
110      * @param slf My alias
111      * @param ctr Counter
112      * @param strt Start
113      * @since 2.7.1
114      * @checkstyle ParameterNumberCheck (5 lines)
115      */
116     DyInbox(final Region reg, final String slf, final Counter ctr,
117         final long strt) {
118         this.region = reg;
119         this.self = slf;
120         this.counter = ctr;
121         this.since = strt;
122     }
123 
124     @Override
125     public long start() throws IOException {
126         final long number = this.counter.incrementAndGet(1L);
127         this.region.table(DyFriends.TBL).put(
128             new Attributes()
129                 .with(DyFriends.RANGE, this.self)
130                 .with(DyFriends.HASH, number)
131                 .with(DyFriends.ATTR_SUBSCRIPTION, true)
132                 .with(DyFriends.ATTR_UPDATED, System.currentTimeMillis())
133                 .with(DyFriends.ATTR_TITLE, "untitled")
134         );
135         Logger.info(this, "bout #%d started by @%s", number, this.self);
136         return number;
137     }

139     // @todo #1:30min  DynamoDBLocal doesn't work with all attributes
140     //  in global index. If we remove this check for a local version,
141     //  most tests fail. I'm not sure how to fix that. I suspect, it's
142     //  a bug in DynamoDBLocal, but I don't even know how to report
143     //  it to them :( Anyway, let's try to investigate and either fix
144     //  property or introduce a better workaround. Pay attention that this
145     //  code works correctly in production.
146     @Override
147     public long unread() throws IOException {
148         long unread = 0L;
149         if (!"1.0-LOCAL".equals(Manifests.read("Netbout-Version"))) {
150             final Iterable<Item> items = this.region.table(DyFriends.TBL)
151                 .frame()
152                 .where(DyFriends.RANGE, this.self)
153                 .through(
154                     new QueryValve()
155                         .withIndexName(DyFriends.INDEX)
156                         .withConsistentRead(false)
157                         .withSelect(Select.SPECIFIC_ATTRIBUTES)
158                         .withAttributesToGet(DyFriends.ATTR_UNREAD)
159                         .withScanIndexForward(false)
160                 );
161             for (final Item item : items) {
162                 if (item.has(DyFriends.ATTR_UNREAD)) {
163                     unread += Long.parseLong(
164                         item.get(DyFriends.ATTR_UNREAD).getN()
165                     );
166                 }
167             }
168         }
169         return unread;
170     }
171 
172     @Override
173     @Loggable(
174         value = Loggable.DEBUG,
175         ignore = Inbox.BoutNotFoundException.class
176         )
177     public Bout bout(final long number) throws Inbox.BoutNotFoundException {
178         final Iterator<Item> items = this.region.table(DyFriends.TBL)
179             .frame()
180             .through(
181                 new QueryValve().withLimit(1)
182                     .withSelect(Select.SPECIFIC_ATTRIBUTES)
183                     .withAttributesToGet(
184                         DyFriends.HASH, DyFriends.RANGE
185                     )
186             )
187             .where(DyFriends.HASH, Conditions.equalTo(number))
188             .where(DyFriends.RANGE, this.self)
189             .iterator();
190         if (!items.hasNext()) {
191             throw new Inbox.BoutNotFoundException(number);
192         }
193         return new DyBout(this.region, items.next(), this.self);
194     }
195 
196     @Override
197     public Pageable<Bout> jump(final long number) {
198         return new DyInbox(this.region, this.self, this.counter, number);
199     }
200 
201     @Override
202     public Iterable<Bout> iterate() {
203         return Iterables.transform(
204             this.region.table(DyFriends.TBL)
205                 .frame()
206                 .where(DyFriends.RANGE, this.self)
207                 .where(
208                     DyFriends.ATTR_UPDATED,
209                     new Condition()
210                         .withComparisonOperator(ComparisonOperator.LT)
211                         .withAttributeValueList(
212                             new AttributeValue().withN(
213                                 Long.toString(this.since)
214                             )
215                         )
216                 )
217                 .through(
218                     new QueryValve()
219                         .withIndexName(DyFriends.INDEX)
220                         .withConsistentRead(false)
221                         .withLimit(Inbox.PAGE)
222                         .withSelect(Select.ALL_PROJECTED_ATTRIBUTES)
223                         .withScanIndexForward(false)
224                 ),
225             new Function<Item, Bout>() {
226                 @Override
227                 public Bout apply(final Item item) {
228                     return new Bout.ReadOnly(
229                         new DyBout(
230                             DyInbox.this.region,
231                             item, DyInbox.this.self
232                         )
233                     );
234                 }
235             }
236         );
237     }
238 
239     @Override
240     public Iterable<Bout> search(final String term) throws IOException {
241         final List<Bout> result = new ArrayList<>(16);
242         for (final Bout bout : this.iterate()) {
243             if (bout.title().contains(term)) {
244                 result.add(bout);
245             } else if (bout.messages().search(term).iterator().hasNext()) {
246                 result.add(bout);
247             }
248         }
249         return result;
250     }
251 
252     /**
253      * Sttc counter.
254      * @return Counter
255      */
256     @Cacheable(forever = true)
257     private static Counter sttc() {
258         try {
259             return new CdSttc(
260                 new ReSttc(
261                     RtSttc.make(
262                         URN.create(Manifests.read("Netbout-SttcUrn")),
263                         Manifests.read("Netbout-SttcToken")
264                     )
265                 )
266             ).counters().get("nb-bout");
267         } catch (final IOException ex) {
268             throw new IllegalStateException(ex);
269         }
270     }
