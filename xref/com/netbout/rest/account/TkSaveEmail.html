27  package com.netbout.rest.account;
28  
29  import com.jcabi.manifests.Manifests;
30  import com.netbout.rest.RqAlias;
31  import com.netbout.spi.Alias;
32  import com.netbout.spi.Base;
33  import java.io.IOException;
34  import java.net.URLEncoder;
35  import org.jasypt.encryption.pbe.StandardPBEStringEncryptor;
36  import org.takes.Request;
37  import org.takes.Response;
38  import org.takes.Take;
39  import org.takes.facets.auth.RqAuth;
40  import org.takes.facets.flash.RsFlash;
41  import org.takes.facets.forward.RsFailure;
42  import org.takes.facets.forward.RsForward;
43  import org.takes.rq.RqForm;
44  import org.takes.rq.RqHref;
45  
46  /**
47   * User account.
48   *
49   * @author Yegor Bugayenko (yegor@teamed.io)
50   * @version $Id: e8272b75f764c1c724228640aa96fde49984be1b $
51   * @since 2.14
52   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
53   */
54  final class TkSaveEmail implements Take {
55  
56      /**
57       * Encryptor.
58       */
59      private static final StandardPBEStringEncryptor ENC =
60          new StandardPBEStringEncryptor();
61  
62      /**
63       * Is Netbout running local?
64       */
65      private final transient boolean local;
66  
67      /**
68       * Base.
69       */
70      private final transient Base base;
71  
72      static {
73          TkSaveEmail.ENC.setPassword(
74              Manifests.read("Netbout-EmailCryptSecret")
75          );
76      }
77  
78      /**
79       * Ctor.
80       * @param bse Base
81       */
82      TkSaveEmail(final Base bse) {
83          this(bse, Manifests.read("Netbout-Version").contains("LOCAL"));
84      }
85  
86      /**
87       * Ctor.
88       * @param bse Base
89       * @param lcl Local Netbout
90       */
91      TkSaveEmail(final Base bse, final boolean lcl) {
92          this.base = bse;
93          this.local = lcl;
94      }
95  
96      @Override
97      public Response act(final Request req) throws IOException {
98          final String email = new RqForm.Smart(
99              new RqForm.Base(req)
100         ).single("email");
101         final Alias alias = new RqAlias(this.base, req).alias();
102         final Response res;
103         if (this.local) {
104             try {
105                 alias.email(email);
106             } catch (final IOException ex) {
107                 throw new RsFailure(ex);
108             }
109             res = new RsForward(
110                 new RsFlash(
111                     String.format("Email changed to \"%s\".", email)
112                 )
113             );
114         } else {
115             final String code = URLEncoder.encode(
116                 TkSaveEmail.ENC.encrypt(
117                     String.format(
118                         "%s:%s:%s",
119                         new RqAuth(req).identity().urn(), alias.name(), email
120                     )
121                 ), "UTF-8"
122             );
123             final String link = String.format(
124                 "%semverify/%s",
125                 new RqHref.Smart(new RqHref.Base(req)).home().bare(), code
126             );
127             final String old;
128             if (alias.email().contains("!")) {
129                 old = alias.email().substring(0, alias.email().indexOf('!'));
130             } else {
131                 old = alias.email();
132             }
133             try {
134                 alias.email(String.format("%s!%s", old, email), link);
135             } catch (final IOException ex) {
136                 throw new RsFailure(ex);
137             }
138             res = new RsForward(
139                 new RsFlash(
140                     // @checkstyle StringLiteralsConcatenationCheck (5 lines)
141                     String.format(
142                         "Email changed to \"%s\". The verification "
143                             + "link sent to this address.",
144                         email
145                     )
146                 )
147             );
148         }
149         return res;
150     }
151 }
