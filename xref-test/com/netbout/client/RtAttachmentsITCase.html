42  /**
43   * Integration case for {@link RtAttachments}.
44   * @author Yegor Bugayenko (yegor@teamed.io)
45   * @version $Id: 4f7f43a2ea95da8d887b19f7c4afda77eea7f017 $
46   * @since 2.4

48  public final class RtAttachmentsITCase {
49  
50      /**
51       * Netbout rule.
52       * @checkstyle VisibilityModifierCheck (3 lines)
53       */
54      @Rule
55      public final transient NbRule rule = new NbRule();
56  
57      /**
58       * RtAttachments can post and read.
59       * @throws Exception If there is some problem inside
60       */
61      @Test
62      public void postsAndReads() throws Exception {
63          final User user = NbRule.get();
64          final Alias alias = user.aliases().iterate().iterator().next();
65          final Inbox inbox = alias.inbox();
66          final Bout bout = inbox.bout(inbox.start());
67          bout.rename(this.getClass().getName());
68          final Attachments attachments = bout.attachments();
69          final String name = "test";
70          attachments.create(name);
71          final Attachment attachment = attachments.get(name);
72          final byte[] data = new byte[Byte.MAX_VALUE - 1];
73          for (int idx = 0; idx < data.length; ++idx) {
74              data[idx] = (byte) idx;
75          }
76          attachment.write(
77              new ByteArrayInputStream(data),
78              Attachment.MARKDOWN,
79              Long.toString(System.currentTimeMillis())
80          );
81          MatcherAssert.assertThat(
82              IOUtils.toByteArray(attachment.read()),
83              Matchers.equalTo(data)
84          );
85          attachments.delete(name);
86      }
87  
88      /**
89       * RtAttachments can obtain author of an attachment.
90       * @throws Exception If there is some problem inside
91       */
92      @Test
93      public void obtainsAuthor() throws Exception {
94          final User user = NbRule.get();
95          final Alias alias = user.aliases().iterate().iterator().next();
96          final Inbox inbox = alias.inbox();
97          final Attachments attachments = inbox.bout(inbox.start())
98              .attachments();
99          final String name = "name";
100         attachments.create(name);
101         final Attachment attachment = attachments.get(name);
102         MatcherAssert.assertThat(
103             attachment.author(), Matchers.equalTo(alias.name())
104         );
105         attachments.delete(name);
106     }
107 
108     /**
109      * RtAttachments can obtain the creation date of an attachment.
110      * @throws Exception If there is some problem inside
111      */
112     @Test
113     public void obtainsCreationDate() throws Exception {
114         final Alias alias = NbRule.get().aliases().iterate().iterator().next();
115         final Inbox inbox = alias.inbox();
116         final Attachments attachments =
117             inbox.bout(inbox.start()).attachments();
118         final long before = System.currentTimeMillis();
119         final String name = "attach-name";
120         attachments.create(name);
121         final Attachment attachment = attachments.get(name);
122         MatcherAssert.assertThat(
123             attachment.date().getTime(), Matchers.greaterThanOrEqualTo(before)
124         );
125         final long after = System.currentTimeMillis();
126         MatcherAssert.assertThat(
127             attachment.date().getTime(), Matchers.lessThanOrEqualTo(after)
128         );
129         attachments.delete(name);
130     }
