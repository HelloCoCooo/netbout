43  /**
44   * Integration case for {@link DyAttachments}.
45   * @author Yegor Bugayenko (yegor@teamed.io)
46   * @version $Id: 21382ff9b4ce85e3c842a73e60fc1ecaf06ca362 $
47   */
48  public final class DyAttachmentsITCase {
49  
50      /**
51       * DyAttachments can create, save and load attachments.
52       * @throws Exception If there is some problem inside
53       */
54      @Test
55      public void createsAndLoadsAttachments() throws Exception {
56          final String alias = "bill";
57          final Aliases aliases =
58              new DyBase().user(new URN("urn:test:840918")).aliases();
59          aliases.add(alias);
60          final Inbox inbox = aliases.iterate().iterator().next().inbox();
61          final Bout bout = inbox.bout(inbox.start());
62          final Attachments attachments = bout.attachments();
63          final String name = "testing-1";
64          attachments.create(name);
65          final Attachment attachment = attachments.get(name);
66          attachment.write(
67              IOUtils.toInputStream("5\u20ac", CharEncoding.UTF_8),
68              MediaType.TEXT_PLAIN,
69              Long.toString(System.currentTimeMillis())
70          );
71          MatcherAssert.assertThat(
72              IOUtils.toString(attachment.read(), CharEncoding.UTF_8),
73              Matchers.containsString("\u20ac")
74          );
75          attachments.delete(name);
76      }
77  
78      /**
79       * DyAttachments will throw exception if the file
80       * length is not between 3-100 characters.
81       * @throws Exception If there is some problem inside
82       */
83      @Test
84      public void throwsExceptionForNameLength() throws Exception {
85          final String alias = "bill1";
86          final Aliases aliases =
87              new DyBase().user(new URN("urn:test:840919")).aliases();
88          aliases.add(alias);
89          final Inbox inbox = aliases.iterate().iterator().next().inbox();
90          final Bout bout = inbox.bout(inbox.start());
91          final Attachments attachments = bout.attachments();
92          final String name = "a";
93          try {
94              attachments.create(name);
95              Assert.fail("The character length exeption was not thrown");
96          } catch (final Attachments.InvalidNameException ex) {
97              MatcherAssert.assertThat(
98                  ex.getMessage(),
99                  Matchers.equalTo(
100                     String.format(
101                         "invalid attachment name length \"%s\". %s%s",
102                         name,
103                         "The length of the attachment name must be ",
104                         "between 3-100 characters"
105                     )
106                 )
107             );
108         }
109     }
